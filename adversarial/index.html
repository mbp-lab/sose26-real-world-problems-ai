What is this notebook about?¶

Adversarial attacks are small intentional changes to input data that cause machine learning models to make wrong predictions, such as adding subtle noise that makes a model classify a panda as a gibbon. Much of the research on adversarial attacks has focused on computer vision. In this notebook, you will explore adversarial attacks for tabular data, which is a less explored but important area in applications such as healthcare and finance.

📌 Note that the code has been generated with the help of an LLM but it has been inspected for the purposes of this exercise. Additional comments have been added to aid your unterstanding! You need to install libraries, such as sklearn, xgboost or art to be able to use the code!

import warnings
warnings.filterwarnings("ignore")

import numpy as np
from sklearn.datasets import load_breast_cancer
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
from sklearn.metrics import accuracy_score
from xgboost import XGBClassifier
from art.estimators.classification import XGBoostClassifier as ART_XGBoostClassifier
from art.attacks.evasion import BoundaryAttack

# Load dataset
# 0 = malignant, 1 = benign
data = load_breast_cancer()
X = data.data.astype(np.float32)
y = data.target.astype(np.int64)
print("Dataset shape:", X.shape)

# Train-test split
# stratify=y preserves class balance
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.25, stratify=y, random_state=42)


# Scale features
# StandardScaler makes features comparable:
# mean = 0, std = 1
scaler = StandardScaler()
X_train = scaler.fit_transform(X_train).astype(np.float32)
X_test = scaler.transform(X_test).astype(np.float32)

# ART requires valid feature ranges
clip_values = (np.min(X_train, axis=0),np.max(X_train, axis=0))

# Train XGBoost model - some standard parameters
xgb = XGBClassifier(
    n_estimators=100,     # number of trees
    max_depth=3,          # tree depth
    learning_rate=0.05,  # smaller = more stable learning
    eval_metric="logloss",
    random_state=42)
xgb.fit(X_train, y_train)
clean_acc = accuracy_score(y_test, xgb.predict(X_test))
print("Accuracy on original data:", round(clean_acc, 4))


# Wrap model with ART
# ART attacks cannot directly use normal XGBoost models.
# The wrapper acts as an adapter between: XGBoost model <-> ART attack library
art_xgb = ART_XGBoostClassifier(
    model=xgb,
    clip_values=clip_values,
    nb_features=X_train.shape[1],
    nb_classes=2)
X_attack = X_test#[:20] # BoundaryAttack is slow, so use only a few samples if you run into runtime issues
y_attack = y_test#[:20]


# Create untargeted BoundaryAttack. Untargeted means that any wrong prediction is counted as a legitimate attack, we don't target a specific class
attack = BoundaryAttack(
    estimator=art_xgb,
    targeted=False,  
    max_iter=20,
    num_trial=20,
    sample_size=20,
    verbose=False)

# Generate adversarial examples
print("\nRunning BoundaryAttack...")
X_adv = attack.generate(x=X_attack)
# Get predictions - ART predict() returns probabilities np.argmax converts them into class labels.
clean_preds = np.argmax(art_xgb.predict(X_attack), axis=1)
adv_preds = np.argmax(art_xgb.predict(X_adv), axis=1)
print("Done!")
